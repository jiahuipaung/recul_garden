```Go
func twoSum(nums []int, target int) []int {
    hashTable := map[int]int{}
    for i, x := range nums {
        if p, ok := hashTable[target - x]; ok {
            return []int{p, i}
        }
        hashTable[x] = i
    }
    return nil
}
```

```Go
func groupAnagrams(strs []string) [][]string {
    // hash表，以排序后字符串为键
    mp := map[string][]string{}
    for _, str := range strs {
        s := []byte(str)  //转为byte方便排序
        sort.Slice(s, func(i, j int) bool {return s[i] < s[j]})
        sortedStr := string(s)
        mp[sortedStr] = append(mp[sortedStr], str)
    }
    ans := make([][]string, 0, len(mp))
    for _, v := range mp {
        ans = append(ans, v)
    }
    return ans
```

## 最长连续序列
#### Tips
以num为键，bool为值，可以快速判断是否出现过该num，遍历两次即可。

```Go
func longestConsecutive(nums []int) int {
    numSet := map[int]bool{}
    for _, num := range nums {
        numSet[num] = true
    }
    longestStreak := 0
    for num := range numSet {
        // 前缀不存在
        if !numSet[num-1] {
            currentNum := num
            currentStreak := 1
            // 再往后连续遍历,若存在
            for numSet[currentNum+1] {
                currentNum++
                currentStreak++
            }
            if longestStreak < currentStreak {
                longestStreak = currentStreak
            }
        }
        // 前缀存在的已经连续处理了
    }
    return longestStreak
}
```

## 无重复字符的最长字串

#### Tips

滑动窗口，最重要的是判断窗口中是否存在重复的元素:用hash表，这里为提高效率用了数组查

```Go
func lengthOfLongestSubstring(s string) (ans int) {
    // 按asic码，保存window中存在的字符
    window := [128]bool{}
    left := 0
    // right依次遍历
    for right, c := range s {
        // 窗口中存在c
        for window[c] {
            // 缩小窗口，至window中没有c
            window[s[left]] = false
            left++
        }
        window[c] = true
        ans = max(ans, right-left+1)
    }
    return
}
```

## 找到字符串中所有字母异位词

#### Tips

滑动窗口，逐个滑动，还是用数组表示字符串中对应字符出现的次数，次数相同就是异位词

**滑动窗口需要满足单调性，当右端点元素进入窗口时，窗口元素和是不能减少的。**

```Go
func findAnagrams(s string, p string) (ans []int) {
    sLen, pLen := len(s), len(p)
    if sLen < pLen {
        return
    }

    var sCount, pCount [26]int
    for i, ch := range p {
        sCount[s[i] - 'a']++   //s前pLen个字符中对应的个数
        pCount[ch - 'a']++   //保存p中各字符的个数
    }

    if sCount == pCount {
        ans = append(ans, 0)
    }

    // 在此逐个滑动，删第0个
    for i, ch := range s[:sLen-pLen] {
        sCount[ch - 'a']--
        sCount[s[i+pLen] - 'a']++
        if sCount == pCount {
            ans = append(ans, i+1)
        }
    }
    return
}
```

# 滑动窗口与双指针

区别：

- 滑动窗口需要维护**窗口内的状态**，这个状态与更新指针位置，和记录答案都有关，找到状态随着指针的移动的最简化的表示以后，就可以高效简洁解决问题。

#### 盛水最多的容器

##### Tips

木桶效应：也可称为短板效应，是讲一只水桶能装多少水取决于它**最短的那块木板**

找两条边都最长的位置，首先就是从两边开始。然后限制条件为短的height，所以谁短谁移动，来找更长的

```Go
func maxArea(height []int) (ans int) {
    left, right := 0, len(height)-1
    for left < right {
        area := (right - left) * min(height[left], height[right])
        ans = max(ans, area)

        if height[left] < height[right] {
            left++
        } else {
            right--
        }
    }
    return
}
```

#### 三数之和

Tips

当暴力枚举的时间复杂度>= n^2时可以考虑排序进行**优化**，优化的性质可以使通过O(1)的时间知道O(n)的信息；

两个优化点；

内部循环为相向双指针

```Go
func threeSum(nums []int) (ans [][]int) {
    // 排序，利用排序后的性质优化，O(1)的时间知道O(n)的信息
    slices.Sort(nums)
    n := len(nums)
    for i, x := range nums[:n-2] {
        // 跳过x的重复数字，重复三元组
        if i > 0 && x == nums[i-1] {
            continue
        }
        // O(1)的时间知道O(n)的信息，因此降了n的时间复杂度
        // 优化1:再往后遍历都>0
        if x+nums[i+1]+nums[i+2] > 0 {
            break
        }
        // 该数x加上最大的还<0，可直接跳过后续
        if x+nums[n-2]+nums[n-1] < 0 {
            continue
        }
        j, k := i+1, n-1
        for j < k {
            s := x + nums[j] + nums[k]
            if s > 0 {
                k--
            } else if s < 0 {
                j++
            } else {
                ans = append(ans, []int{x, nums[j], nums[k]})
                for j++; j < k && nums[j] == nums[j-1]; j++{} //跳过重复数字
                for k--; k > j && nums[k] == nums[k+1]; k--{}
            }
        }
    }
    return
}
```

```Go
func trap(height []int) int {
    left, right := 0, len(height)-1
    preMax, sufMax := 0, 0
    for left < right {
        // 前缀最大值 = max(之前的前缀最大值，当前值）
        preMax = max(preMax, height[left])
        sufMax = max(sufMax, height[right])

            left++
        } else {
            ans += sufMax - height[right]
```

## 快排（分治法）

_快速排序__的最坏运行情况是 O(n²)，比如说顺序数列的快排。但它的平摊期望时间是 O(nlogn)，且 O(nlogn) 记号中隐含的常数因子很小，比复杂度稳定等于 O(nlogn) 的__归并排序__要小很多。所以，对绝大多数顺序性较弱的随机数列而言，快速排序总是优于归并排序。_

```Go
func quickSort(nums []int, l, r int) {
    if l >= r {
        return
    }
    pivot := nums[l]
    i, j := l-1, r+1
    for i < j {
        for i++; nums[i]<pivot; i++ {} // 快速过滤元素
        for j--; nums[j]>pivot; j-- {}
        if (i < j) {
            nums[i], nums[j] = nums[j], nums[i]
        }
    }
    quickSort(nums, l, j)
    quickSort(nums, j+1, r)

}
```

## 桶排序

桶排序大概的过程：

- 设置一个定量的数组当作空桶；
    
- 遍历输入数据，并且把元素一个一个放到对应的桶里去；
    
- 对每个不是空的桶进行排序；(可以使用其他排序算法或桶递归)
    
- 从不是空的桶里把排好序的数据拼接起来。
    

```Go
func bucketSort(numArr []int,bucketSize int){

    minValue := numArr[0]
    maxValue := numArr[0]

    for i := 0; i<len(numArr); i++ {
        if minValue > numArr[i] {
            minValue = numArr[i]
        }
        if maxValue < numArr[i] {
            maxValue = numArr[i]
        }
    }

    // 桶切片初始化
    bucketCount := make([][]int,(maxValue - minValue) / bucketSize +1)

    // 数据入桶
    for i := 0; i<len(numArr); i++ {
        bucketCount[(numArr[i] - minValue) / bucketSize] = append(bucketCount[(numArr[i] - minValue) / bucketSize] , numArr[i])
    }

    key := 0
    // 对每个桶进行排序，这里使用冒泡排序
    for _ , bucket := range bucketCount {
        if (len(bucket) <= 0) { continue; }
        insertionSort(bucket);
        for _ , value := range bucket {
            numArr[key] = value
            key = key+1
        }
    }
    return
}
```

  

# 贪心

**原理**：**每一步都采取当前状态下在局部看来最优的选择**，以期通过局部最优解最终获得全局最优解。

**适用条件：**

1. 贪心选择性质：在每一个阶段做出的局部最优选择不会影响全局最优解的产生。（局部最优选择是最终全局最优解的一部分）
    
    1. **尝试构造反例**：在局部最优选择下，看看是否能够构造一个场景，在该场景中贪心选择导致错误解。
        
    2. **选择策略和最优解的对比**：从理论上分析当前的局部最优策略是否可以保证每一步的选择都能够包含在全局最优解中。
        
2. 最优子结构：一个问题的最优解可以由其子问题的最优解构成
    
    1. **递归****验证**：将问题递归地拆解成更小的子问题，验证子问题的最优解能否组合成原问题的最优解。
        
    2. **动态规划解法**：通常具有最优子结构的问题，可以用动态规划来解决。
        

两种基本贪心策略：

- 从最小/最大开始贪心，优先考虑最小/最大的数，从小到大/从大到小贪心。在此基础上，衍生出了反悔贪心。
    
- 从最左/最右开始贪心，思考第一个数/最后一个数的贪心策略，把 n 个数的原问题转换成n−1 个数（或更少）的子问题。

2. ### **腐烂的橘子**
    

A：多源DFS；遍历记录两个数据：所有fresh的个数（作为遍历结束的条件）、所有腐烂的坐标（作为多源遍历对象）；每轮遍历的都是上一轮刚腐烂的，类似层序遍历；

```Go
type pair struct{x, y int}
var dirs = []pair{{-1, 0}, {1, 0}, {0, -1}, {0, 1}}
func orangesRotting(grid [][]int) (ans int) {
    m, n := len(grid), len(grid[0])
    fresh := 0
    q := []pair{}
    for i, row := range grid {
        for j, x := range row {
            if x == 1 { // 统计新鲜橘子个数
                fresh++                
            } else if x == 2 {
                q = append(q, pair{i, j})
            }
        }
    }
    for fresh > 0 && len(q) > 0 {
        ans++
        tmp := q  // 每轮开始时腐烂的橘子都不一样，因此需要创建新的并保存，否则会死循环，q初始化为空，下一轮只遍历上一轮刚腐烂掉的
        q = []pair{}
        for _, p := range tmp {  // 遍历一开始就腐烂的橘子
            for _, d := range dirs { // 朝四个方向，处理一轮
                i, j := p.x+d.x, p.y+d.y
                if 0 <= i && i < m && 0 <= j && j < n && grid[i][j] == 1 {
                    fresh--
                    grid[i][j] = 2 // 变腐烂
                    q = append(q, pair{i, j}) // 保存新变腐烂的橘子
                }
            }

        }
    }

    if fresh > 0 {
        return -1
    }
    return ans
}
```

## 贪心算法

1. ### 买股票最佳时机
    

A：从左到右枚举卖出价格 prices[i]，那么要想获得最大利润，我们需要知道第 i 天之前，股票价格的最小值是什么，也就是从 prices[0] 到 prices[i−1] 的最小值，把它作为买入价格，这可以用一个变量 minPrice 维护。

请注意，minPrice 维护的是 prices[i] 左侧元素的最小值。

由于只能买卖一次，所以在遍历中，维护 prices[i]−minPrice 的最大值，就是答案。

3. ### 跳跃游戏2️⃣
    

A：求的是跳到最后一个点所需的最小跳跃数。按建桥的思路

为什么只用遍历到n-2，因为如果确实到这个点了，只需要建桥，不用再跳了，如果没到，那就表示可以直接到n-1了。

之前局限点是怀疑我在这个点跳了，那后面有没有可能有能跳更远的。但是用nextRight维护了最远的，所以保证了覆盖

#### 为什么是最优的？

1. **贪心策略**：
    
    1. 每次选择跳跃的目标时，都选择可以到达的最远位置。这样的选择是局部最优的，确保每次跳跃都尽量扩大覆盖范围。
        
2. **覆盖所有情况**：
    
    1. 通过不断更新 `curRight` 和 `nextRight`，算法确**保在跳跃的每一步都涵盖了所有可能的跳跃路径**，从而不会漏掉任何潜在的跳跃方式。
        
3. **最小跳跃次数**：
    
    1. 由于在每一步都尽量跳到最远的可达位置，所以最终的跳跃次数是最小的。这是因为在到达 `curRight` 后，跳跃的边界总是更新为 `nextRight`，意味着在尽量少的跳跃中探索所有可能的路径。
        

#### 保证最优性

- **无后效性**：每次的选择只依赖于当前的状态，而不受之前选择的影响。每次到达 `curRight` 后，下一步的最优选择（即选择下一个最远的 `nextRight`）不会影响前面的决定。
    
- **覆盖路径**：算法确保不会遗漏任何到达终点的路径，因为它在每一步都在更新可以到达的最远位置。
    
- **跳跃顺序的选择**：贪心算法在此场景中是合适的，因为每次选择都是基于当前的信息来最大化未来的选择。

1. ### 爬楼梯
    

因为要解决的问题都是「从 0 爬到 i」，所以定义 dfs(i) 表示从 0 爬到 i 有多少种不同的方法。

分类讨论：

- 如果最后一步爬了 1 个台阶，那么我们得先爬到 i−1，要解决的问题缩小成：从 0 爬到 i−1 有多少种不同的方法。
    
- 如果最后一步爬了 2 个台阶，那么我们得先爬到 i−2，要解决的问题缩小成：从 0 爬到 i−2 有多少种不同的方法。
    

由于这两种方法是互相独立的（爬的台阶个数不同），所以根据加法原理，从 0 爬到 i 的方法数等于这两种方法数之和，即

dfs(i)=dfs(i−1)+dfs(i−2)

递归边界：dfs(0)=1, dfs(1)=1。从 0 爬到 0 有一种方法，即原地不动。从 0 爬到 1 有一种方法，即爬 1 个台阶。

递归入口：dfs(n)，也就是答案。
